Private Sub HideDuplicateTitleLabels()
    Dim c As Control

    On Error Resume Next

    For Each c In Me.Controls
        If TypeName(c) = "Label" Then
            If Trim$(CStr(c.Caption)) = "LAMBDA Function Editor" Then
                If c.Name <> "lblAppTitle" Then
                    c.Caption = vbNullString
                    c.Visible = False
                    c.Height = 0
                End If
            End If
        End If
    Next c

    On Error GoTo 0
End Sub

Private Sub RefreshList()
    Dim names As Collection
    Dim x As Variant
    Dim re As Object
    Dim filterText As String
    Dim useFilter As Boolean
    Dim totalCount As Long

    On Error GoTo FilterError

    mLoading = True
    lstNames.Clear

    filterText = Trim$(txtFilter.Text)

    If Len(filterText) > 0 Then
        Set re = CreateObject("VBScript.RegExp")
        re.Pattern = filterText
        re.IgnoreCase = True
        re.Global = False
        useFilter = True
    End If

    Set names = GetLambdaNames(ActiveWorkbook)

    For Each x In names
        totalCount = totalCount + 1

        If Not useFilter Then
            lstNames.AddItem CStr(x)
        ElseIf re.Test(CStr(x)) Then
            lstNames.AddItem CStr(x)
        End If
    Next x

    On Error GoTo 0

    If useFilter Then
        lblStatus.Caption = CStr(lstNames.ListCount) & " of " & CStr(totalCount) & " LAMBDA function(s) shown."
    Else
        lblStatus.Caption = CStr(lstNames.ListCount) & " LAMBDA function(s) found."
    End If

    mLoading = False
    Exit Sub

FilterError:
    lblStatus.Caption = "Invalid regex filter: " & Err.Description
    mLoading = False
End Sub


Private Sub ClearEditor()
    mLoading = True

    mCurrentName = vbNullString
    txtName.Text = vbNullString
    txtComment.Text = vbNullString
    txtFormula.Text = "=LAMBDA(" & vbCrLf & "    x," & vbCrLf & "    x" & vbCrLf & ")"
    txtTestFormula.Text = vbNullString
    txtResult.Text = vbNullString

    mDirty = False
    mLoading = False
    txtName.SetFocus
End Sub

Private Sub LoadLambda(ByVal lambdaName As String)
    Dim nm As Name

    Set nm = GetLambdaName(ActiveWorkbook, lambdaName)
    If nm Is Nothing Then Exit Sub

    mLoading = True

    mCurrentName = nm.Name
    txtName.Text = nm.Name
    txtFormula.Text = nm.RefersTo

    On Error Resume Next
    txtComment.Text = nm.Comment
    On Error GoTo 0

    txtTestFormula.Text = "=" & nm.Name & "()"
    txtResult.Text = vbNullString

    mDirty = False
    mLoading = False
End Sub

Private Function ConfirmDiscardChanges() As Boolean
    If Not mDirty Then
        ConfirmDiscardChanges = True
    Else
        ConfirmDiscardChanges = MsgBox("Discard unsaved changes?", vbQuestion + vbYesNo, "LAMBDA Editor") = vbYes
    End If
End Function

    mDirty = True

End Sub

Private Sub HandleListSelection()
    Dim i As Long
    Dim selectedCount As Long
    Dim selectedIndex As Long
    Dim selectedName As String

    If mLoading Then Exit Sub
    If mHandlingListSelection Then Exit Sub

    mHandlingListSelection = True
    On Error GoTo CleanUp

    selectedIndex = -1

    For i = 0 To lstNames.ListCount - 1
        If lstNames.Selected(i) Then
            selectedCount = selectedCount + 1
            If selectedIndex = -1 Then selectedIndex = i
        End If
    Next i

    If selectedCount = 0 Then GoTo CleanUp

    If selectedCount > 1 Then
        lblStatus.Caption = CStr(selectedCount) & " LAMBDA functions selected."
        GoTo CleanUp
    End If

    selectedName = CStr(lstNames.List(selectedIndex))

    If StrComp(selectedName, mCurrentName, vbTextCompare) = 0 And Not mDirty Then GoTo CleanUp

    If Not ConfirmDiscardChanges Then GoTo CleanUp

    LoadLambda selectedName

CleanUp:
    mHandlingListSelection = False
End Sub

Private Sub SaveCurrentLambda()
    On Error GoTo Fail

    SaveLambdaName ActiveWorkbook, txtName.Text, txtFormula.Text, txtComment.Text
    mCurrentName = CleanNameText(txtName.Text)
    mDirty = False

    RefreshList
    SelectNameInList mCurrentName
    LoadLambda mCurrentName

    lblStatus.Caption = "Saved " & mCurrentName
    Exit Sub

Fail:
    lblStatus.Caption = Err.Description
    MsgBox Err.Description, vbExclamation, "Save failed"
End Sub

Private Sub SelectNameInList(ByVal lambdaName As String)
    Dim i As Long

    For i = 0 To lstNames.ListCount - 1
        If StrComp(CStr(lstNames.List(i)), lambdaName, vbTextCompare) = 0 Then
            lstNames.ListIndex = i
            Exit Sub
        End If
    Next i
End Sub


Private Sub cmdDelete_Click()
    Dim i As Long
    Dim count As Long
    Dim lambdaName As String
    Dim msg As String

    For i = 0 To lstNames.ListCount - 1
        If lstNames.Selected(i) Then count = count + 1
    Next i

    If count = 0 Then
        lambdaName = CleanNameText(txtName.Text)
        If Len(lambdaName) = 0 Then Exit Sub

        msg = "Delete " & lambdaName & "?"
        If MsgBox(msg, vbQuestion + vbYesNo, "Delete LAMBDA") <> vbYes Then Exit Sub

        On Error GoTo FailSingle
        DeleteLambdaName ActiveWorkbook, lambdaName
        RefreshList
        ClearEditor
        lblStatus.Caption = "Deleted " & lambdaName
        Exit Sub
    End If

    If count = 1 Then
        For i = 0 To lstNames.ListCount - 1
            If lstNames.Selected(i) Then
                lambdaName = CStr(lstNames.List(i))
                Exit For
            End If
        Next i

        If MsgBox("Delete " & lambdaName & "?", vbQuestion + vbYesNo, "Delete LAMBDA") <> vbYes Then Exit Sub

        On Error GoTo FailSingle
        DeleteLambdaName ActiveWorkbook, lambdaName
        RefreshList
        ClearEditor
        lblStatus.Caption = "Deleted " & lambdaName
        Exit Sub
    End If

    ' First confirmation for true batch deletes.
    If MsgBox("You are about to delete " & CStr(count) & " LAMBDA functions. Continue?", _
              vbExclamation + vbYesNo, "Confirm batch delete") <> vbYes Then Exit Sub

    ' Second stronger confirmation for true batch deletes.
    If MsgBox("This action cannot be undone. Delete all selected functions?", _
              vbCritical + vbYesNo, "Final confirmation") <> vbYes Then Exit Sub

    On Error GoTo FailMulti

    For i = lstNames.ListCount - 1 To 0 Step -1
        If lstNames.Selected(i) Then
            DeleteLambdaName ActiveWorkbook, CStr(lstNames.List(i))
        End If
    Next i

    RefreshList
    ClearEditor
    lblStatus.Caption = "Deleted " & CStr(count) & " LAMBDA functions."
    Exit Sub

FailSingle:
    lblStatus.Caption = Err.Description
    MsgBox Err.Description, vbExclamation, "Delete failed"
    Exit Sub

FailMulti:
    lblStatus.Caption = Err.Description
    MsgBox Err.Description, vbExclamation, "Batch delete failed"
End Sub




Private Sub cmdValidate_Click()
    ValidateCurrentLambda
End Sub

Private Sub ValidateCurrentLambda()
    Dim v As Variant

    v = TryEvaluateFormula(txtFormula.Text, ActiveWorkbook)

    If IsError(v) Then
        txtResult.Text = ValueToText(v)
        lblStatus.Caption = "Formula did not evaluate cleanly. This may still be valid if it expects arguments."
    Else
        txtResult.Text = ValueToText(v)
        lblStatus.Caption = "Formula evaluates."
    End If
End Sub

Private Sub cmdTest_Click()
    Dim v As Variant
    Dim expr As String

    expr = Trim$(txtTestFormula.Text)

    If Len(expr) = 0 Then
        MsgBox "Enter a test formula, for example =MyLambda(1,2).", vbInformation, "Test LAMBDA"
        Exit Sub
    End If

    If mDirty Then SaveCurrentLambda

    v = TryEvaluateFormula(expr, ActiveWorkbook)
    txtResult.Text = ValueToText(v)
    lblStatus.Caption = "Test completed."
End Sub

Private Sub txtFormula_KeyDown(ByVal KeyCode As MSForms.ReturnInteger, ByVal Shift As Integer)
    If Shift = 2 And KeyCode = vbKeyS Then
        KeyCode = 0
        SaveCurrentLambda
    ElseIf Shift = 2 And KeyCode = vbKeyR Then
        KeyCode = 0
        txtFormula.Text = FormatLambdaFormula(txtFormula.Text)
        txtFormula.SelStart = Len(txtFormula.Text)
    ElseIf Shift = 2 And KeyCode = vbKeyM Then
        KeyCode = 0
        cmdMinify_Click
    ElseIf Shift = 3 And KeyCode = vbKeyV Then
        KeyCode = 0
        cmdVisualize_Click
    ElseIf KeyCode = vbKeyTab Then
        KeyCode = 0
        InsertAtCursor txtFormula, "    "
    End If
End Sub

Private Sub txtTestFormula_KeyDown(ByVal KeyCode As MSForms.ReturnInteger, ByVal Shift As Integer)
    If KeyCode = vbKeyReturn Then
        KeyCode = 0
        cmdTest_Click
    End If
End Sub

Private Sub InsertAtCursor(ByVal tb As MSForms.TextBox, ByVal s As String)
    Dim p As Long

    p = tb.SelStart
    tb.Text = Left$(tb.Text, p) & s & Mid$(tb.Text, p + tb.SelLength + 1)
    tb.SelStart = p + Len(s)
End Sub
